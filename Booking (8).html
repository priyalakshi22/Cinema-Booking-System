package com.cinema.model; import java.time.LocalDateTime; public class Showtime { private int id; private Movie movie; private String hall; private LocalDateTime time; private int availableSeats; public Showtime(int id, Movie movie, String hall, LocalDateTime time, int availableSeats) { this.id = id; this.movie = movie; this.hall = hall; this.time = time; this.availableSeats = availableSeats; } // Getters and setters public int getId() { return id; } public Movie getMovie() { return movie; } public String getHall() { return hall; } public LocalDateTime getTime() { return time; } public int getAvailableSeats() { return availableSeats; } public void setAvailableSeats(int availableSeats) { this.availableSeats = availableSeats; } }
