package com.cinema.service; import com.cinema.model.*; import java.time.LocalDateTime; import java.util.ArrayList; import java.util.List; public class CinemaService { private List movies = new ArrayList<>(); private List showtimes = new ArrayList<>(); private List customers = new ArrayList<>(); private List bookings = new ArrayList<>(); private int movieIdCounter = 1; private int showtimeIdCounter = 1; private int customerIdCounter = 1; private int bookingIdCounter = 1; public CinemaService() { System.out.println("CinemaService initialized with in-memory storage"); } // ================= MOVIES ================= public List getAllMovies() { return new ArrayList<>(movies); } public Movie getMovieById(int id) { return movies.stream() .filter(m -> m.getId() == id) .findFirst() .orElse(null); } public Movie addMovie(String title, int duration, double price) { Movie movie = new Movie(movieIdCounter++, title, duration, price); movies.add(movie); return movie; } // ✅ NEW: Update movie by ID public boolean updateMovie(Movie updatedMovie) { for (int i = 0; i < movies.size(); i++) { Movie movie = movies.get(i); if (movie.getId() == updatedMovie.getId()) { // Update all fields movie.setTitle(updatedMovie.getTitle()); movie.setDuration(updatedMovie.getDuration()); movie.setPrice(updatedMovie.getPrice()); // If you have more fields, update them here too return true; } } return false; } // ================= SHOWTIMES ================= public Showtime addShowtime(int movieId, String hall, LocalDateTime time, int availableSeats) { Movie movie = getMovieById(movieId); if (movie == null) return null; Showtime showtime = new Showtime( showtimeIdCounter++, movie, hall, time, availableSeats ); showtimes.add(showtime); return showtime; } public List getAllShowtimes() { return new ArrayList<>(showtimes); } // ================= CUSTOMERS ================= public Customer addCustomer(String name, String email, String phone) { Customer customer = new Customer( customerIdCounter++, name, email, phone ); customers.add(customer); return customer; } public List getAllCustomers() { return new ArrayList<>(customers); } // ================= BOOKINGS ================= public Booking createBooking(int customerId, int showtimeId, int seatCount) { Customer customer = customers.stream() .filter(c -> c.getId() == customerId) .findFirst() .orElse(null); Showtime showtime = showtimes.stream() .filter(s -> s.getId() == showtimeId) .findFirst() .orElse(null); if (customer == null || showtime == null) return null; if (showtime.getAvailableSeats() < seatCount) return null; showtime.setAvailableSeats( showtime.getAvailableSeats() - seatCount ); Booking booking = new Booking( bookingIdCounter++, customer, showtime, seatCount ); bookings.add(booking); return booking; } public List getAllBookings() { return new ArrayList<>(bookings); } }
