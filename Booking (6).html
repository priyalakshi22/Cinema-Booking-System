package com.cinema.model; public class Movie { private int id; private String title; private int duration; // in minutes private double price; public Movie(int id, String title, int duration, double price) { this.id = id; this.title = title; this.duration = duration; this.price = price; } // Getters and Setters public int getId() { return id; } public void setId(int id) { this.id = id; } public String getTitle() { return title; } public void setTitle(String title) { this.title = title; } public int getDuration() { return duration; } public void setDuration(int duration) { this.duration = duration; } public double getPrice() { return price; } public void setPrice(double price) { this.price = price; } @Override public String toString() { return "Movie [id=" + id + ", title=" + title + ", duration=" + duration + " minutes, price=$" + price + "]"; } }
